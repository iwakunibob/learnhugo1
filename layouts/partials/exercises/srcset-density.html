{{- /* get image resource */ -}} {{- $img := .page.Resources.GetMatch .src -}} {{- /* height to width ratio of src image */ -}} {{- /* numbers must be converted to float before division */ -}} {{- if not $img -}} {{- warnf "no image resource found at %s%s" .page.RelPermalink .src -}} {{- end -}} {{- /* set variables */ -}} {{- $densities := .densities | default (slice 1 2) -}} {{- $src := "" -}} {{- $srcset := slice -}} {{- range $index, $density := $densities -}} {{- $width := mul $.width $density -}} {{- if gt $width $img.Width -}} {{- warnf "Source image %s is not wide enough to resize to %dpx and %dx density" $img.RelPermalink $width $density -}} {{- end -}} {{- $resized := $img.Resize (printf "%dx" $width) -}} {{- $srcset := $srcset | append (printf "%s %d" $resized.RelPermalink $density) -}} {{- if eq $density 1 -}} {{- $src = $resized -}} {{- end -}} {{- end -}}
